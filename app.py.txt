import gradio as gr
from diffusers import AutoPipelineForText2Image
import torch

MODEL_ID = "stabilityai/sdxl-turbo"

pipe = AutoPipelineForText2Image.from_pretrained(
    MODEL_ID,
    torch_dtype=torch.float16,
    variant="fp16"
)

pipe = pipe.to("cuda")

def generate_image(
    prompt,
    negative_prompt="",
    width=1024,
    height=1024,
    steps=4,
    guidance_scale=0.0
):
    image = pipe(
        prompt=prompt,
        negative_prompt=negative_prompt,
        width=width,
        height=height,
        num_inference_steps=steps,
        guidance_scale=guidance_scale
    ).images[0]

    return image

with gr.Blocks(
    title="Limited AI Image Generator"
) as demo:

    gr.Markdown(
        """
        # Limited AI
        ### Text-to-Image Generator
        """
    )

    with gr.Row():
        with gr.Column():
            prompt = gr.Textbox(
                label="Prompt",
                lines=5,
                placeholder="A futuristic city at sunset..."
            )

            negative_prompt = gr.Textbox(
                label="Negative Prompt",
                lines=2
            )

            width = gr.Slider(
                512,
                1536,
                value=1024,
                step=64,
                label="Width"
            )

            height = gr.Slider(
                512,
                1536,
                value=1024,
                step=64,
                label="Height"
            )

            steps = gr.Slider(
                1,
                30,
                value=4,
                step=1,
                label="Steps"
            )

            generate_btn = gr.Button(
                "Generate",
                variant="primary"
            )

        with gr.Column():
            output_image = gr.Image(
                label="Generated Image"
            )

    generate_btn.click(
        fn=generate_image,
        inputs=[
            prompt,
            negative_prompt,
            width,
            height,
            steps
        ],
        outputs=output_image
    )

demo.queue(max_size=20)
demo.launch()
